0 12 * * 1-5 $PYTHON $SCRIPT noon >> $LOG 2>&1

# 收盘流程（生成明日计划）- 周一到周五 15:30
30 15 * * 1-5 $PYTHON $SCRIPT evening >> $LOG 2>&1

# 周复盘 - 每周日 20:00
0 20 * * 0 $PYTHON $SCRIPT weekly >> $LOG 2>&1
